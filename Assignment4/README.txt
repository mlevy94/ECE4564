HOW TO:
	mineserver.py - program to be launched on server RPi. Minecraft must first be running.
	mineclient.py - program to launch on client RPis. mineserver.py must first be running.
ARGUMENTS:
	mineserver.py
	mineclient.py [server ip]
DEPENDENCIES:
	Python Version: 3.5+
	modules:
		aiocoap
		mcpi
		RPi.GPIO
